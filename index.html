def kalkulator(a, b, dzialanie):
    if dzialanie == "+":
        return a + b
    elif dzialanie == "-":
        return a - b
    elif dzialanie == "*":
        return a * b
    elif dzialanie == "/":
        return a / b
    else:
        return "Nieznana operacja"

print(kalkulator(4, 2, "*"))

from datetime import datetime

data = input("Podaj swoj dat urodzin (RRRR-MM-DD): ")
urodziny = datetime.strptime(data, "%Y-%m-%d")
dzis = datetime.now()
roznica = urodziny.replace(year=dzis.year) - dzis

if roznica.days < 0:
    urodziny = urodziny.replace(year=dzis.year + 1)
    roznica = urodziny - dzis

print("Do Twoich urodzin zostao:", roznica.days, "dni!")
